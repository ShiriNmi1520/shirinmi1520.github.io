<?php
$url = "API網址"
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURL_OPT_HTTPHEADER, array( //設置header
    'HEADER1': 'value',
    'HEADER2': 'value',
));
$output = curl_exec($ch); //取得回傳
curl_close($ch);
?>

發送請求(POST) ¶

<?php
$url = "API網址"
$payload = json_encode( array( "customer"=> $data ) ); //post json資料
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt( $ch, CURLOPT_POSTFIELDS, $payload );
curl_setopt( $ch, CURLOPT_HTTPHEADER, array('Content-Type:application/json')); //設置header
$output = curl_exec($ch); //取得回傳
curl_close($ch);
?>
